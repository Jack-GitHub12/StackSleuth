Quick Start

Express.js Integration

npm install @stacksleuth/backend-agent @stacksleuth/core
import express from 'express';
import { BackendAgent } from '@stacksleuth/backend-agent';

const app = express();
const agent = new BackendAgent({
  enabled: true,
  projectId: 'your-project-id',
  apiKey: 'your-api-key'
});

// Start monitoring
agent.startMonitoring();
app.use(agent.middleware());

app.get('/api/users', async (req, res) => {
  const users = await getUsersFromDatabase();
  res.json(users);
});

app.listen(3000);

Vue.js Integration

npm install @stacksleuth/vue-agent
import { createApp } from 'vue';
import { VueAgent } from '@stacksleuth/vue-agent';
import App from './App.vue';

const app = createApp(App);

const agent = new VueAgent({
  enabled: true,
  trackComponents: true,
  trackRouter: true
});

app.use(agent);
agent.startMonitoring();

app.mount('#app');

Django Integration

pip install stacksleuth-django
# settings.py
INSTALLED_APPS = [
    # ... your apps
    'stacksleuth.django',
]

MIDDLEWARE = [
    'stacksleuth.django.middleware.StackSleuthMiddleware',
    # ... your middleware
]

STACKSLEUTH = {
    'ENABLED': True,
    'PROJECT_ID': 'your-project-id',
    'API_KEY': 'your-api-key',
    'SAMPLE_RATE': 0.1,
}

API Service Setup

npm install @stacksleuth/api
import { StackSleuthAPI } from '@stacksleuth/api';

const api = new StackSleuthAPI();
api.start(3000);

// API Documentation: http://localhost:3000/api-docs
// Health Check: http://localhost:3000/health
